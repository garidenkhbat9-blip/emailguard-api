Developer API Reference

Endpoint

POST https://api.yourdomain.com/api/validate

cURL Request

curl -X POST http://localhost:3000/api/validate \
  -H "Content-Type: application/json" \
  -d '{"email":"team@techcorp.com"}'

Node.js / Axios

const axios = require('axios');

async function verifyEmail(email) {
  const response = await axios.post('http://localhost:3000/api/validate', { email });
  
  // Block user if Risk Score is high
  if (response.data.action === 'Block') {
    throw new Error('High risk email detected.');
  }
  return response.data;
}
